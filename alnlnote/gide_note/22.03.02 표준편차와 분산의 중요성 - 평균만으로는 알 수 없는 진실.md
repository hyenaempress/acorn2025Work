def average_trap_demo():
    """평균의 함정을 보여주는 극단적 예제"""
    
    print("🕳️ 평균의 함정 - 극단적 사례")

    # 시나리오 2: 매우 불안정한 그룹  
    unstable_group = [10, 40, 60, 80, 110]  # 평균 60
    
    # 시나리오 3: 양극화된 그룹
    polarized_group = [30, 30, 60, 90, 90]  # 평균 60
    
    scenarios = {
        "안정적 그룹": stable_group,
        "불안정 그룹": unstable_group,
        "양극화 그룹": polarized_group
    }
    
    print("📊 세 그룹 모두 평균은 60점:")
    print("-" * 35)
    
    for name, data in scenarios.items():
        mean_val = np.mean(data)
        std_val = np.std(data)
        min_val, max_val = np.min(data), np.max(data)
        
        print(f"\n{name}:")
        print(f"   데이터: {data}")
        print(f"   평균: {mean_val:.1f}점")
        print(f"   표준편차: {std_val:.2f}")
        print(f"   범위: {min_val}~{max_val}점")
        print(f"   해석: ", end="")
        
        if std_val < 1:
            print("매우 안정적, 예측 가능")
        elif std_val < 15:
            print("보통 수준의 변동성")
        elif std_val < 25:

## 🧪 실습: 두 반의 성적 비교

### 📊 **강사님 예제 완전 재현**

```python
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

# 한글 폰트 설정
plt.rc('font', family='Malgun Gothic')
plt.rcParams['axes.unicode_minus'] = False

def teacher_example_reproduction():
    """강사님 예제 완전 재현"""
    
    print("🎓 강사님 예제 재현 - 두 반의 성적 비교")
    print("=" * 50)
    
    # 랜덤 시드 고정 (강사님이 강조하신 부분)
    np.random.seed(42)
    print("🔧 랜덤 시드 42로 고정 (재현 가능한 결과)")
    
    # 목표 설정
    target_mean = 60
    std_dev_small = 10  # 1반: 표준편차 작음 (고른 실력)
    std_dev_large = 20  # 2반: 표준편차 큼 (실력 차이 큼)
    
    print(f"\n📋 설정:")
    print(f"   목표 평균: {target_mean}점")
    print(f"   1반 표준편차: {std_dev_small} (실력이 고름)")
    print(f"   2반 표준편차: {std_dev_large} (실력 차이 큼)")
    
    # 데이터 생성 과정 (강사님이 설명하신 단계별 처리)
    print(f"\n🔄 데이터 생성 과정:")
    
    # 1단계: 기본 정규분포 생성
    class1_raw = np.random.normal(loc=target_mean, scale=std_dev_small, size=100)
    class2_raw = np.random.normal(loc=target_mean, scale=std_dev_large, size=100)
    print("   1단계: 정규분포로 기본 데이터 생성")
    
    # 2단계: 평균 보정 (강사님이 "살짝 손을 대는" 부분)
    class1_adj = (class1_raw - np.mean(class1_raw)) + target_mean
    class2_adj = (class2_raw - np.mean(class2_raw)) + target_mean
    print("   2단계: 평균을 정확히 60점으로 보정")
    
    # 3단계: 정수화 및 범위 제한 (현실적인 점수로)
    class1 = np.clip(np.round(class1_adj), 10, 100).astype(int)
    class2 = np.clip(np.round(class2_adj), 10, 100).astype(int)
    print("   3단계: 정수화 및 10~100점 범위로 제한")
    
    # 결과 출력
    print(f"\n📊 생성된 데이터 (처음 10개):")
    print(f"   1반: {class1[:10]}")
    print(f"   2반: {class2[:10]}")
    
    return class1, class2, target_mean

# 실제 실습 실행
class1, class2, target_mean = teacher_example_reproduction()
```

def calculate_and_analyze(class1, class2):
    """통계값 계산 및 상세 분석"""
    
    print("\n📊 통계값 계산 및 분석")
    print("=" * 35)
    
    # 기본 통계값 계산
    stats_data = {
        "1반": {
            "mean": np.mean(class1),
            "std": np.std(class1, ddof=1),  # 표본 표준편차
            "var": np.var(class1, ddof=1),  # 표본 분산
            "min": np.min(class1),
            "max": np.max(class1),
            "range": np.max(class1) - np.min(class1)
        },
        "2반": {
            "mean": np.mean(class2),
            "std": np.std(class2, ddof=1),
            "var": np.var(class2, ddof=1),
            "min": np.min(class2),
            "max": np.max(class2),
            "range": np.max(class2) - np.min(class2)
        }
    }
    
    # 결과 출력
    print("📋 통계값 비교:")
    print("-" * 55)
    print(f"{'구분':<10} {'평균':<8} {'표준편차':<10} {'분산':<10} {'범위':<8}")
    print("-" * 55)
    
    for class_name, stats in stats_data.items():
        print(f"{class_name:<10} "
              f"{stats['mean']:<8.1f} "
              f"{stats['std']:<10.2f} "
              f"{stats['var']:<10.2f} "
              f"{stats['min']}-{stats['max']}")
    
    # 강사님 스타일의 해석
    print(f"\n🎯 강사님 스타일 해석:")
    
    std_diff = abs(stats_data['2반']['std'] - stats_data['1반']['std'])
    var_diff = abs(stats_data['2반']['var'] - stats_data['1반']['var'])
    
    interpretations = [
        f"📊 평균은 두 반 모두 {stats_data['1반']['mean']:.1f}점으로 거의 동일",
        f"📏 하지만 표준편차는 {std_diff:.2f}만큼 차이 (약 {std_diff/stats_data['1반']['std']*100:.0f}% 차이)",
        f"📈 분산은 {var_diff:.2f}만큼 차이 (약 {var_diff/stats_data['1반']['var']*100:.0f}% 차이)",
        "",
        "🔍 실제 의미:",
        f"   • 1반: 실력이 고른 집단 (표준편차 {stats_data['1반']['std']:.1f})",
        f"   • 2반: 실력 편차가 큰 집단 (표준편차 {stats_data['2반']['std']:.1f})",
        "",
        "💡 교육적 시사점:",
        "   • 1반: 전체적으로 비슷한 수준, 일률적 교육 가능",
        "   • 2반: 수준별 맞춤 교육 필요, 개별 지도 중요"
    ]
    
    for interpretation in interpretations:
        print(f"   {interpretation}")
    
    return stats_data

# 분석 실행
stats_result = calculate_and_analyze(class1, class2)

def detailed_analysis_report(class1, class2):
    """상세 분석 리포트 생성"""
    
    print("\n📋 상세 분석 리포트")
    print("=" * 30)
    
    # 기본 통계값
    stats1 = {
        'mean': np.mean(class1),
        'std': np.std(class1, ddof=1),
        'var': np.var(class1, ddof=1),
        'cv': np.std(class1, ddof=1) / np.mean(class1) * 100,  # 변동계수
        'q25': np.percentile(class1, 25),
        'q75': np.percentile(class1, 75),
        'iqr': np.percentile(class1, 75) - np.percentile(class1, 25)
    }
    
    stats2 = {
        'mean': np.mean(class2),
        'std': np.std(class2, ddof=1),
        'var': np.var(class2, ddof=1),
        'cv': np.std(class2, ddof=1) / np.mean(class2) * 100,
        'q25': np.percentile(class2, 25),
        'q75': np.percentile(class2, 75),
        'iqr': np.percentile(class2, 75) - np.percentile(class2, 25)
    }
    
    print("📊 종합 통계 비교:")
    print("-" * 70)
    print(f"{'항목':<15} {'1반':<15} {'2반':<15} {'차이':<15}")
    print("-" * 70)
    
    comparisons = [
        ('평균', stats1['mean'], stats2['mean']),
        ('표준편차', stats1['std'], stats2['std']),
        ('분산', stats1['var'], stats2['var']),
        ('변동계수(%)', stats1['cv'], stats2['cv']),
        ('1사분위수', stats1['q25'], stats2['q25']),
        ('3사분위수', stats1['q75'], stats2['q75']),
        ('사분위범위', stats1['iqr'], stats2['iqr'])
    ]
    
    for item, val1, val2 in comparisons:
        diff = abs(val2 - val1)
        print(f"{item:<15} {val1:<15.2f} {val2:<15.2f} {diff:<15.2f}")
    
    # 해석 및 결론
    print(f"\n🔍 심층 분석:")
    
    analyses = [
        "📊 중심 경향성:",
        f"   • 두 반의 평균 차이: {abs(stats2['mean'] - stats1['mean']):.2f}점 (거의 동일)",
        f"   • 평균만으로는 두 반이 비슷해 보임",
        "",
        "📏 변산성 (핵심!):",
        f"   • 표준편차 차이: {abs(stats2['std'] - stats1['std']):.2f}점",
        f"   • 2반이 1반보다 {stats2['std']/stats1['std']:.1f}배 더 변동성이 큼",
        f"   • 변동계수: 1반 {stats1['cv']:.1f}% vs 2반 {stats2['cv']:.1f}%",
        "",
        "🎯 실무적 함의:",
        "   • 1반: 예측 가능한 집단, 표준화된 교육 방법 효과적",
        "   • 2반: 개별 차이 큰 집단, 맞춤형 교육 전략 필요",
        "",
        "⚠️ 위험성 평가:",
        f"   • 1반: 낮은 위험성 (안정적 성과 예상)",
        f"   • 2반: 높은 위험성 (성과 예측 어려움)",

        "💡 강사님 메시지의 핵심:",
        "   '평균만으로는 절대 알 수 없는 중요한 정보가 있다!'"

    for analysis in analyses:
        print(f"   {analysis}")
