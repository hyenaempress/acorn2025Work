## 📋 프로젝트 개요

이 프로젝트는 Django 프레임워크를 사용하여 성별에 따른 커피 브랜드 선호도 차이를 카이제곱 검정으로 분석하는 웹 애플리케이션입니다.

- 카이제곱 독립성 검정 수행

```sql
CREATE DATABASE coffeedb;

-- 테이블 생성

    rnum INT PRIMARY KEY AUTO_INCREMENT,
```

```python
from django.shortcuts import render, redirect
from django.http import HttpResponse
from django.db.models import Count
from .models import Survey
import scipy.stats as stats
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
plt.rc('font', family='Malgun Gothic')

def survey_main(request):
    """메인 페이지"""
    return render(request, 'index.html')

def survey_form(request):
    """설문 입력 폼 페이지"""
    return render(request, 'coffee/survey_form.html')

def survey_result(request):
    """설문 결과 및 카이제곱 검정 결과 출력"""
    surveys = Survey.objects.all()
    
    # 성별×브랜드 교차표를 위해 집계
    agg = (
        Survey.objects
        .values('gender', 'co_survey')
        .annotate(cnt=Count('rnum'))
    )
    
    # 성별/브랜드 축 만들기
    genders = sorted({row['gender'] for row in agg})
    brands = sorted({row['co_survey'] for row in agg})
    
    result_text = "데이터가 부족합니다."
    chi2 = None
    dof = None
    decision = None
    
    if len(genders) >= 2 and len(brands) >= 2:
        # 관측도수 행렬 생성
        idx_g = {g:i for i,g in enumerate(genders)}
        idx_b = {b:j for j,b in enumerate(brands)}
        r, c = len(genders), len(brands)
        obs = [[0 for _ in range(c)] for __ in range(r)]
        
        for row in agg:
            i = idx_g[row['gender']]
            j = idx_b[row['co_survey']]
            obs[i][j] = row['cnt']
        
        # 기대도수 계산
        row_sum = [sum(obs[i]) for i in range(r)]
        col_sum = [sum(obs[i][j] for i in range(r)) for j in range(c)]
        total = sum(row_sum)
        
        expected = [[(row_sum[i] * col_sum[j]) / total for j in range(c)] for i in range(r)]
        
        # 카이제곱 통계량 계산
        chi2_val = 0.0
        for i in range(r):
            for j in range(c):
                e = expected[i][j]
                o = obs[i][j]
                if e > 0:
                    chi2_val += (o - e) ** 2 / e
        
        chi2 = round(chi2_val, 4)
        dof = (r - 1) * (c - 1)
        
        # 유의수준 0.05 임계값 테이블
        critical_005 = {
            1: 3.8415, 2: 5.9915, 3: 7.8147, 4: 9.4877, 5: 11.0705,
            6: 12.5916, 7: 14.0671, 8: 15.5073, 9: 16.9190, 10: 18.3070,
        }
        crit = critical_005.get(dof, None)
        
        if crit is None:
            decision = f"자유도 {dof}에 대한 0.05 임계값이 표에 없습니다. (χ²={chi2})"
        else:
            if chi2 > crit:
                decision = f"χ²={chi2} > 임계값 {crit:.4f} → 귀무가설 기각: 성별에 따른 선호 브랜드에 **차이가 있음**"
            else:
                decision = f"χ²={chi2} ≤ 임계값 {crit:.4f} → 귀무가설 채택: 성별에 따른 선호 브랜드 **차이가 없음**"
        
        # 교차분할표 구성
        crosstab = {
            'headers': ['성별 \\ 브랜드'] + brands,
            'rows': [
                [g] + [obs[idx_g[g]][idx_b[b]] for b in brands]
                for g in genders
            ],
            'row_totals': row_sum,
            'col_totals': col_sum,
            'total': total,
        }
        
        return render(request, 'coffee/survey_result.html', {
            'surveys': surveys,
            'decision': decision,
            'chi2': chi2,
            'dof': dof,
            'alpha': 0.05,
            'crosstab': crosstab,
        })
    
    # 데이터 부족 시 기본 출력
    return render(request, 'coffee/survey_result.html', {
        'surveys': surveys,
        'decision': result_text,
        'chi2': chi2,
        'dof': dof,
        'alpha': 0.05,
        'crosstab': None,
    })

def survey_process(request):
    """설문 데이터 처리 및 저장"""
    if request.method != 'POST':
        return render(request, 'coffee/survey_form.html')
    
    gender = request.POST.get('gender')
    age = request.POST.get('age')
    co_survey = request.POST.get('co_survey')
    
    Survey.objects.create(
        gender=gender,
        age=int(age) if age else None,
        co_survey=co_survey,
    )
    return redirect('survey_result')
```

## 📊 카이제곱 검정 이론

### 가설 설정
- **귀무가설 (H₀)**: 성별과 커피 브랜드 선호도는 서로 독립이다 (관련이 없다)
- **대립가설 (H₁)**: 성별과 커피 브랜드 선호도는 서로 독립이 아니다 (관련이 있다)

### 검정 통계량
```
χ² = Σ[(관측도수 - 기대도수)² / 기대도수]
```

### 자유도
```
df = (행의 개수 - 1) × (열의 개수 - 1)
```

### 판정 기준
- **α = 0.05** (유의수준 5%)
- **χ² > 임계값**: 귀무가설 기각 → 관련성 있음
- **χ² ≤ 임계값**: 귀무가설 채택 → 관련성 없음

### 2. 데이터 검증
```python
def validate_survey_data(gender, age, co_survey):
    """설문 데이터 유효성 검사"""
    errors = []
    
    if not gender or gender not in ['남', '여']:
        errors.append('올바른 성별을 선택해주세요.')
    
    if not age or int(age) not in [10, 20, 30, 40, 50]:
        errors.append('올바른 나이대를 선택해주세요.')
    
    valid_brands = ['스타벅스', '커피빈', '이디야', '탐앤탐스']
    if not co_survey or co_survey not in valid_brands:
        errors.append('올바른 커피 브랜드를 선택해주세요.')
    
    return errors
```
