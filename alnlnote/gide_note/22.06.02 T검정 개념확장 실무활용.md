### ğŸ“Š 2. ë…ë¦½í‘œë³¸ tê²€ì • (Independent samples t-test)

    female_stats = {

        'mean': np.mean(female_spending),
        'std': np.std(female_spending, ddof=1)
    }

    print(f"   í‘œë³¸ í¬ê¸°: {male_stats['n']}")

    print(f"   í‘œë³¸ í¬ê¸°: {female_stats['n']}")
    print(f"   í‰ê· : {female_stats['mean']:,.0f}ì›")
    print(f"   í‘œì¤€í¸ì°¨: {female_stats['std']:,.0f}ì›")

    else:

    print(f"\nğŸ“Š ë…ë¦½í‘œë³¸ tê²€ì • ê²°ê³¼:")

    # íš¨ê³¼í¬ê¸° (Cohen's d) ê³„ì‚°

    # íš¨ê³¼í¬ê¸° í•´ì„
    if abs(cohens_d) < 0.2:
        effect_size = "ì‘ìŒ"
    elif abs(cohens_d) < 0.5:
        effect_size = "ì¤‘ê°„"
    elif abs(cohens_d) < 0.8:
        effect_size = "í¼"
    else:
        effect_size = "ë§¤ìš° í¼"
    
    print(f"íš¨ê³¼í¬ê¸° í•´ì„: {effect_size}")
    
    # ê²°ê³¼ í•´ì„
    alpha = 0.05
