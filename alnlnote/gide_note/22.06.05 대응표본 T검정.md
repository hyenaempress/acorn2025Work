```python

    # 2ë‹¨ê³„: ì •ê·œì„± ê²€ì • (ì°¨ì´ê°’)

        use_parametric = True

        use_parametric = False

    if use_parametric:

        print(f"   T-statistic: {t_stat:.6f}")

    else:

        print(f"   Wilcoxon statistic: {wilcox_stat:.1f}")
        test_method = "Wilcoxon ê²€ì •"
    
    print(f"   P-value: {p_value:.6f}")
    print(f"   ììœ ë„: {len(before) - 1}")
```
